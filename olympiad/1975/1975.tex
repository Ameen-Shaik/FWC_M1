\begin{center}
{\Large \textbf{Seventeenth International Olympiad, 1975}}
\end{center}

\textbf{1975/1.}

Let $x_i,y_i \ (i=1,2,\ldots,n)$ be real numbers such that

$$
x_1\geq x_2\geq\cdots\geq x_n
\quad \text{and} \quad
y_1\geq y_2\geq\cdots\geq y_n.
$$

Prove that, if $z_1,z_2,\ldots,z_n$ is any permutation of $y_1,y_2,\ldots,y_n$, then

$$
\sum_{i=1}^{n}(x_i-y_i)^2 \leq \sum_{i=1}^{n}(x_i-z_i)^2.
$$

\vspace{0.4cm}

\textbf{1975/2.}

Let $a_1,a_2,a_3,\ldots$ be an infinite increasing sequence of positive integers. Prove that for every $p\geq1$ there are infinitely many $a_m$ which can be written in the form

$$
a_m=xa_p+ya_q
$$

with $x,y$ positive integers and $q>p$.

\vspace{0.4cm}

\textbf{1975/3.}

On the sides of an arbitrary triangle $ABC$, triangles $ABR$, $BCP$, $CAQ$ are constructed externally with $\angle CBP=\angle CAQ=45^\circ$, $\angle BCP=\angle ACQ=30^\circ$, and $\angle ABR=\angle BAR=15^\circ$. Prove that $\angle QRP=90^\circ$ and $QR=RP$.

\textbf{1975/4.}

When $4444^{4444}$ is written in decimal notation, the sum of its digits is $A$. Let $B$ be the sum of the digits of $A$. Find the sum of the digits of $B$. ($A$ and $B$ are written in decimal notation.)

\vspace{0.4cm}

\textbf{1975/5.}

Determine, with proof, whether or not one can find 1975 points on the circumference of a circle with unit radius such that the distance between any two of them is a rational number.

\vspace{0.4cm}

\textbf{1975/6.}

Find all polynomials $P$, in two variables, with the following properties:

\begin{enumerate}
\item[(i)] for a positive integer $n$ and all real $t,x,y$

$$
P(tx,ty)=t^nP(x,y)
$$

(that is, $P$ is homogeneous of degree $n$),

\item[(ii)] for all real $a,b,c$,

$$
P(b+c,a)+P(c+a,b)+P(a+b,c)=0,
$$

\item[(iii)] $P(1,0)=1$.
\end{enumerate}
